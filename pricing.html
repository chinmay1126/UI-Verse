  position:absolute; inset:0;
  background:#ddd; border-radius:999px;
  cursor:pointer; transition:0.3s;
}
.toggle-switch input:checked + .toggle-track { background:#eb6835; }
.toggle-track::before {
  content:''; position:absolute;
  width:18px; height:18px; background:#fff;
  border-radius:50%; top:3px; left:3px; transition:0.3s;
}
.toggle-switch input:checked + .toggle-track::before { transform:translateX(20px); }
 
function switchBilling() {
  const isAnnual = document.getElementById('billingToggle').checked;
  document.querySelectorAll('.mini-price').forEach(el => {
    el.textContent = isAnnual ? el.dataset.annual : el.dataset.monthly;
  });
}
